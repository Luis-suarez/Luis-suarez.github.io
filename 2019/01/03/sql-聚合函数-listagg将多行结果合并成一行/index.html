聚合函数 listagg将多行结果合并成一行

select goods_no,
master_takegoods_no,
master,
car_no,
listagg(to_char(wz_info), ‘;’) within group(ORDER BY wz_info) as wz_info
from (select td.goods_no,
td.master_takegoods_no,
tdd.id,
td.master,
(tdd.item_name || ‘,’ || tdd.material || ‘,’ || tdd.spec || ‘,’ ||
tdd.producer || ‘,’ || decode(tdd.metering_type,’1’,’理计’,’2’,’过磅’,’3’,’抄码’) || ‘,’ ||
tdd.physical_weight || ‘,’ || tdd.receivable_number || ‘,’ ||tdd.receivable_weight) wz_info,
wm_concat(tf.car_no) car_no
from t_deliver_goods td
left join t_deliver_detailed tdd
on td.goods_no = tdd.business_no
left join t_forecast_middle_deliver tf
on td.goods_no = tf.goods_no
where td.return_state in ‘0’
and td.fh_state in ‘0’
and tdd.receivable_number > 0
group by td.goods_no,
tdd.id,
td.master_takegoods_no,
td.master,
tdd.item_name,
tdd.material,
tdd.spec,
tdd.producer,
tdd.physical_weight,
tdd.metering_type,
tdd.receivable_number,
tdd.receivable_weight)
group by goods_no, master_takegoods_no, master, car_no
